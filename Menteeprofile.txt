// select thong tin co ban cua mentee và số lượng người mentee đang theo dõi
SELECT 
    u.id,
    u.name,
    j.name as jobtitle,
    u.location,
    u.role,
    (SELECT COUNT(*) 
     FROM Follow 
     WHERE follower_id = 'U3') as total_followers
FROM Userr u 
JOIN Jobtitle j on u.jobtitle = j.id
WHERE u.id = 'U3';
